# 数据劫持
+ 数据劫持：分为对象劫持和数组劫持
+ 不管哪种劫持，都会在被劫持的对象新增一个__ob__属性，值是Observer对象，它负责保存收集的依赖Dep
+ 对象劫持：遍历data对象树，递归创建Observer对象，遍历对象的所有key，使用Object.defineProperty添加getter和setter，getter负责依赖收集，setter负责检查数据是否修改，生成虚拟dom，对比虚拟dom，更新真实dom
+ 数组劫持：对于数组类型的对象，会修改它的原型对象为新的原型，新的原型的原型仍是数组的原型Array.prototype，新的原型对push、pop等方法做了数据劫持，在原方法的基础上，做了2件事：1、对数组新增的元素进行数据劫持。2、发布更新事件，触发视图更新
+ 数据劫持不会修改data的引用指向，在原对象的基础上增加getter和setter
+ Vue实例也通过Object.defineProperty做了代理，从而可以直接访问data中的属性
+ 数组索引没有做数据劫持，数组元素的属性仍然进行了数据劫持。
+ 为什么不对数组索引做数据劫持：vue2使用Object.defineProperty做数据劫持，数组实际上是key为0,1,2...的对象，理论上可以做，但索引通常变化频繁，代价较高，另外新增数组元素仍无法实现数据劫持，直接通过length删除和新增数组元素时，不会触发setter
+ 数组索引的更新：通过Vue.set或vm.$set，底层仍是调用数组的splice方法。
+ 新增属性：通过Vue.set或vm.$set，不能在Vue实例或根data对象中新增属性，内部使用Object.defineProperty在target中新增getter和setter，然后发布新增事件，触发视图更新

function sameVnode (a, b) {
  return (
    a.key === b.key &&                       // key 必须相等
    a.tag === b.tag &&                       // 标签相同
    a.isComment === b.isComment &&           // 注释节点标识相同
    isDef(a.data) === isDef(b.data)          // 都有或都没有 data（用于判断可否是同个组件）

  )

// 主要的 children diff 逻辑
function updateChildren (parentElm, oldCh, newCh) {
  let oldStartIdx = 0, newStartIdx = 0
  let oldEndIdx = oldCh.length - 1, newEndIdx = newCh.length - 1
  let oldStartVnode = oldCh[0], oldEndVnode = oldCh[oldEndIdx]
  let newStartVnode = newCh[0], newEndVnode = newCh[newEndIdx]
  let oldKeyToIdx, idxInOld, vnodeToMove, refElm

  while (oldStartIdx <= oldEndIdx && newStartIdx <= newEndIdx) {
    if (!oldStartVnode) {
      oldStartVnode = oldCh[++oldStartIdx]
    } else if (!oldEndVnode) {
      oldEndVnode = oldCh[--oldEndIdx]
    }
      // 下面是典型的双端对比：newStart 与 oldStart / oldEnd；newEnd 与 oldEnd / oldStart
    else if (sameVnode(oldStartVnode, newStartVnode)) {
      patchVnode(oldStartVnode, newStartVnode)
      oldStartVnode = oldCh[++oldStartIdx]
      newStartVnode = newCh[++newStartIdx]
    } else if (sameVnode(oldEndVnode, newEndVnode)) {
      patchVnode(oldEndVnode, newEndVnode)
      oldEndVnode = oldCh[--oldEndIdx]
      newEndVnode = newCh[--newEndIdx]
    } else if (sameVnode(oldStartVnode, newEndVnode)) {
      // 旧头与新尾相同，说明节点移动到尾部
      patchVnode(oldStartVnode, newEndVnode) //递归处理子节点
      nodeOps.insertBefore(
        parentElm,
        oldStartVnode.elm,
        nodeOps.nextSibling(oldEndVnode.elm) //移动真实dom,nodeOps.insertBefore内部调用了document的api移动真实dom
      )
      oldStartVnode = oldCh[++oldStartIdx]
      newEndVnode = newCh[--newEndIdx]
    } else if (sameVnode(oldEndVnode, newStartVnode)) {
      // 旧尾与新头相同，说明节点移动到头部
      patchVnode(oldEndVnode, newStartVnode)
      nodeOps.insertBefore(parentElm, oldEndVnode.elm, oldStartVnode.elm)
      oldEndVnode = oldCh[--oldEndIdx]
      newStartVnode = newCh[++newStartIdx]
    } else {
      // **重点：key 查找机制**
      if (oldKeyToIdx === undefined) {
        oldKeyToIdx = createKeyToOldIdx(oldCh, oldStartIdx, oldEndIdx) //双端对比失败，使用keyToIdx
      }
      idxInOld = oldKeyToIdx[newStartVnode.key]
      if (idxInOld == null) {
        // 新节点是全新的，直接创建
        createElm(newStartVnode, parentElm, oldStartVnode.elm)
      } else {
        // 复用 idxInOld 对应的旧节点
        vnodeToMove = oldCh[idxInOld]
        if (sameVnode(vnodeToMove, newStartVnode)) {
          patchVnode(vnodeToMove, newStartVnode)
          oldCh[idxInOld] = undefined
          nodeOps.insertBefore(parentElm, vnodeToMove.elm, oldStartVnode.elm)
        } else {
          // key 相同但实际上 not sameVnode，则作为全新节点处理
          createElm(newStartVnode, parentElm, oldStartVnode.elm)
        }
      }
      newStartVnode = newCh[++newStartIdx]
    }
  }

  // 处理多余的新增或多余的旧节点（略）
}

// 辅助：生成 key -> 旧 children 索引 的映射
function createKeyToOldIdx (children, beginIdx, endIdx) {
  const map = {}
  for (let i = beginIdx; i <= endIdx; ++i) {
    const key = children[i] && children[i].key
    if (key != null) map[key] = i
  }
  return map
}
function patchVnode (
  oldVnode,
  vnode,
  insertedVnodeQueue, // 用于收集插入钩子
  ownerArray,          // 如果 vnode 来自一个数组（siblings），用于在 removeOnly 时打标记
  index,
  removeOnly
) {
  // 1. 如果两个 VNode 完全相同，直接返回
  if (oldVnode === vnode) {
    return
  }

  const elm = vnode.elm = oldVnode.elm  // 复用旧的 DOM 元素
  const oldCh = oldVnode.children
  const ch = vnode.children
  const data = vnode.data

  // 2. 调用 prepatch 钩子（用户自定义或模块钩子）
  if (data && data.hook && data.hook.prepatch) {
    data.hook.prepatch(oldVnode, vnode)
  }

  // 3. 如果新 vnode 没有 text（即要么有 children，要么都没有 text/children）
  if (isUndef(vnode.text)) {
    // 3.1 旧 vnode 有 children，新 vnode 也有 children → 深度 diff
    if (oldCh && ch && oldCh !== ch) {
      updateChildren(elm, oldCh, ch, insertedVnodeQueue, removeOnly)

    // 3.2 新 vnode 有 children，旧 vnode 没有 → 新增所有子节点
    } else if (ch) {
      addVnodes(elm, null, ch, 0, ch.length - 1, insertedVnodeQueue)

    // 3.3 旧 vnode 有 children，新 vnode 没没有 → 移除所有旧子节点
    } else if (oldCh) {
      removeVnodes(oldCh, 0, oldCh.length - 1)

    // 3.4 既没有 children，也没有 text，但旧 vnode 可能有 text → 清空文本
    } else if (oldVnode.text) {
      nodeOps.setTextContent(elm, '')
    }
  }
  // 4. 如果新 vnode 定义了 text，且与旧 vnode.text 不同 → 更新文本
  else if (oldVnode.text !== vnode.text) {
    nodeOps.setTextContent(elm, vnode.text)
  }

  // 5. 调用 postpatch 钩子
  if (data && data.hook && data.hook.postpatch) {
    data.hook.postpatch(oldVnode, vnode)
  }
}

# Vue生命周期
## 创建
+ 创建空的Vue实例，初始化声明周期函数，初始化render函数的createElement函数(vm.$createElement)，执行beforeCreate
+ 1、初始化data，将data进行数据劫持；2、初始化计算属性，将计算属性封装成Watch对象加到依赖的data属性的依赖列表；3、初始化监视属性，如果监视属性的immediate为true，执行一次监视属性的handler函数；4、初始化methods。执行created
+ 如果没指定el，只会执行到created,Vue创建完毕后，通过vm.$mount函数传入el执行时，才回继续执行后续流程

## 挂载
+ 判断是否指定el属性和template属性，如果指定了el没指定template，将el的outerHTML(el对应的整个dom树的HTML)作为template编译成render函数，如果指定了el和template，将template模板编译成render函数，保存到vm.$options.render
+ 执行beforeMount之后，执行render函数，创建虚拟dom（vnode）,执行vm._update，对新旧虚拟dom执行patch，将vm.$el修改为新的真实dom，执行mounted
+ 如果同时指定了el和template，el对应的dom会被template替换掉（仍会patch对比，尽可能复用）
+ template只能一个非template标签的根元素
+ vm.$el中保存了渲染阶段生成的真实dom(初始化的dom)

## 更新
+ 修改数据，先执行beforeUpdate，再根据新的数据生成新的虚拟dom，与旧的虚拟dom进行比较，如果有变化，更新页面真实dom，执行updated
+ beforeUpdate执行时，数据和页面尚不同步

### 虚拟dom
+ 虚拟dom的作用：通过虚拟dom的对比，便于实现真实dom的复用，最小化修改真实dom。虚拟dom是js对象，虽然render函数会重建整个虚拟dom，性能代价仍然较低。新旧dom对比使用双端队列+hash表，每一次只对比同层的dom，避免了整个dom树的对比
+ dom的复用只发生在同一层级，无法跨越父子层级复用

## 路由跳转流程(Vue3)
+ vue2对应vue-router3，vue3对应vue-router4
1. 解析路由(resolve)：根据目标路由的path或name，从选项对象中匹配对应的路由对象，对于多级路由，会将所有父级路由都匹配出来
2. 执行路由守卫(runGuards)：执行当前路由的离开守卫、目标路由的进入守卫
3. 路由跳转（finalizeNavigation）：修改url，保存浏览器历史记录，修改当前路由对象（currentRoute），处理滚动策略。其中修改当前路由对象，会触发router-link组件重渲染
4. router-link重渲染：router-link通过计算属性监听了currentRoute，因此会触发重渲染。对于多级路由，会有多个层级的router-link，通过provied和inject depth，来给不同层级的router-link编号，从0开始，父级inject depth后，加1，provide给子。每个router-link使用depth作为索引读取匹配的路由对象，从而将多级路由和多级router-link匹配，每个router-link负责渲染对应路由的组件
+ vue-router还会监听popstate事件，该事件回退按钮或history.back时触发，用于实现回退不整页刷新
+ 页面初始化、直接修改url时，会触发整页刷新和页面初始化流程，router.install中，会获取window.location，去除base后，得到的uri，push(uri)
+ nginx部署时，如果是history模式，需要配置try_files $uri $uri/index.html index.html，会依次尝试try_files后面的地址，都找不到时，返回index.html，当访问路由路径时，ng返回的是index.html，然后执行页面初始化流程，跳转到对应路由

## 指令
